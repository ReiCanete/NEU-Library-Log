rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Comprehensive admin check
    // Explicitly whitelists the user email with a case-insensitive match
    function isAdmin() {
      return request.auth != null && (
        request.auth.token.email.lower() == 'reiangelo.canete@neu.edu.ph' ||
        request.auth.token.email.lower().matches('.*25-14294-549.*') ||
        (exists(/databases/$(database)/documents/users/$(request.auth.uid)) && 
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin')
      );
    }

    // Visitors collection - Publicly readable/creatable for the Kiosk
    match /visits/{visitId} {
      allow create, read: if true; 
      allow update, delete: if isAdmin();
    }

    // Users collection - Publicly readable/creatable for registration
    match /users/{userId} {
      allow read, create: if true; 
      allow update: if isAdmin() || request.auth.uid == userId; 
    }

    // Blocklist - Publicly readable for the Kiosk to check status
    match /blocklist/{blockId} {
      allow read: if true; 
      allow write: if isAdmin();
    }

    // Announcements - Publicly readable for the Kiosk alerts
    match /announcements/{announcementId} {
      allow read: if true; 
      allow write: if isAdmin();
    }

    // Settings - Publicly readable for Kiosk capacity checks
    match /settings/{settingId} {
      allow read: if true; 
      allow write: if isAdmin();
    }
  }
}
