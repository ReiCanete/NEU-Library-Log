rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Absolute authorization for designated institutional staff
    function isAdmin() {
      return request.auth != null && (
        request.auth.token.email.lower() == 'reiangelo.canete@neu.edu.ph' ||
        request.auth.token.email.lower().matches('.*25-14294-549.*') ||
        (exists(/databases/$(database)/documents/users/$(request.auth.uid)) && 
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin')
      );
    }

    // Visitors collection - Public entry (create), Public monitoring (read), Admin management
    match /visits/{visitId} {
      allow create: if true;
      allow read: if true; 
      allow update, delete: if isAdmin();
    }

    // Users collection - Registration and Admin lookup
    match /users/{userId} {
      allow read: if true;
      allow create: if true; 
      allow update: if isAdmin() || request.auth.uid == userId; 
    }

    // Blocklist - Restricted list management
    match /blocklist/{blockId} {
      allow read: if true; 
      allow write: if isAdmin();
    }

    // Announcements - Public display, Admin management
    match /announcements/{announcementId} {
      allow read: if true; 
      allow write: if isAdmin();
    }

    // Settings - System configuration
    match /settings/{settingId} {
      allow read: if true; 
      allow write: if isAdmin();
    }
  }
}
