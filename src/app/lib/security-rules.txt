rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Function to check if the user is an admin
    function isAdmin() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    // Function to check if user belongs to the @neu.edu.ph domain
    function isNeuUser() {
      return request.auth.token.email.endsWith('@neu.edu.ph');
    }

    // Visitors collection - logging entries
    match /visits/{visitId} {
      allow create: if true; // Allowed from kiosk (even anonymous ID scan)
      allow read, update, delete: if isAdmin();
    }

    // Users collection - profiles and roles
    match /users/{userId} {
      allow read: if request.auth != null && (request.auth.uid == userId || isAdmin());
      allow create: if request.auth != null && isNeuUser();
      allow update: if isAdmin(); // Only admins can change roles or block status
    }

    // Blocklist collection
    match /blocklist/{blockId} {
      allow read: if true; // Kiosk needs to check this (or check user doc)
      allow write: if isAdmin();
    }

    // Announcements collection
    match /announcements/{announcementId} {
      allow read: if true; // Publicly readable on kiosk
      allow write: if isAdmin();
    }

    // Settings collection
    match /settings/{settingId} {
      allow read: if true; // Publicly readable for capacity check
      allow write: if isAdmin();
    }
  }
}
